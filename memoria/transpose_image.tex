    \foreach \i in {0,...,3} {
       \foreach \j in {0,...,3} {
           \pgfmathparse{int(Mod(\i,4))}
           \let\ii\pgfmathresult

           \let\jj\pgfmathresult
           \node at (\i+0.5,3-\j+0.5) {$(\ii, \jj)$};
       }
    }
